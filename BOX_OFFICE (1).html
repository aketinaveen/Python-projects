from sklearn.feature_extraction.text import CountVectorizer

# Create an instance of CountVectorizer
vectorizer = CountVectorizer()

# Fit the vectorizer on the 'genres' column
vectorizer.fit(df['genres'])

# Transform the 'genres' column into a set of binary columns
features = vectorizer.transform(df['genres']).toarray()

# Get the names of the new binary columns
genres = vectorizer.get_feature_names_out()

# Add the new binary columns to the DataFrame (checking for existence)
for i, name in enumerate(genres):
    if name not in df.columns:
        df[name] = features[:, i]

# Drop the original 'genres' column
df.drop('genres', axis=1, inplace=True)
